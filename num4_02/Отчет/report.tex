\section{Решение задачи 2}
$$
DCG = \sum_{i=1}^{N} = \dfrac{rel_i}{\log_2 (i+1)}
$$
$$
DCG_1 = \sum_{i=1}^5 \dfrac{rel_i}{\log_2 (i+1)} = \dfrac{4}{\log_2 2} + \dfrac{2}{\log_2 3} + \dfrac{3}{\log_2 4} + \dfrac{1}{\log_2 5} + \dfrac{2}{\log_2 6} = 7.96
$$
$$
DCG_2 = \dfrac{3}{\log_2 2} + \dfrac{2}{\log_2 3} + \dfrac{4}{\log_2 4} + \dfrac{4}{\log_2 5} + \dfrac{4}{\log_2 6} = 9.53
$$
$$
idealDCG_1 = \dfrac{4}{\log_2 2} + \dfrac{3}{\log_2 3} + \dfrac{2}{log_2 4} + \dfrac{2}{\log_2 5} + \dfrac{1}{\log_2 6} = 8.14
$$
$$
idealDCG_2 = \dfrac{4}{\log_2 2} + \dfrac{4}{\log_2 3} + \dfrac{4}{\log_2 4} + 
\dfrac{3}{\log_2 5} + \dfrac{2}{\log_2 6} = 10.58
$$
$$
NDCG_1 = \dfrac{DCG_1}{idealDCG_1} = \dfrac{7.96}{8.14} = 0.977
$$
$$
NDCG_2 = \dfrac{9.53}{10.58} = 0.900
$$
$$
NDCG_1 > NDCG_2 \Rightarrow \textbf{первая система лучше}
$$
